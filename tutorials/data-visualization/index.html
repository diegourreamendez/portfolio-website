Introduction

Visualization is not decoration — it is the primary channel through which a hydrologist communicates uncertainty, spatial heterogeneity, and temporal trends to reviewers, decision-makers, and the public. A figure that misrepresents confidence intervals or compresses a scale to hide variability is as misleading as an incorrect equation.

This tutorial covers two complementary workflows: matplotlib for publication-ready static figures destined for journals (controlled layout, precise typography, vector export), and Plotly for interactive web figures where the reader can explore data themselves — zoom, hover, filter by trace.

Static → Journal

matplotlib + rcParams, PDF/SVG export, LaTeX text rendering, tight layout control.

Interactive → Web

Plotly graph_objects, hover tooltips, range sliders, dual axes, animated transitions.

Spatial → Maps

Cartopy projections, pcolormesh for gridded data, Plotly Scattergeo for station networks.

Matplotlib Foundations

The first step for any matplotlib figure destined for publication is setting rcParams globally. This separates style configuration from figure logic — swap the style dict and every figure in your paper updates consistently.

01_matplotlib_style.py 
import matplotlib.pyplot as plt
import matplotlib.ticker as ticker
import numpy as np

# ── Global style (apply once at top of script) ──────────────────
plt.rcParams.update({
    "figure.facecolor":  "#0a1628",
    "axes.facecolor":    "#0a1628",
    "axes.edgecolor":    "#3c494c",
    "axes.labelcolor":   "#d7e3fc",
    "xtick.color":       "#859397",
    "ytick.color":       "#859397",
    "text.color":        "#d7e3fc",
    "grid.color":        "#1f2a3d",
    "grid.linestyle":    "--",
    "grid.linewidth":    0.6,
    "font.family":       "monospace",
    "figure.dpi":        150,
    "savefig.bbox":      "tight",
    "savefig.transparent": False,
})

# ── Build figure ────────────────────────────────────────────────
rng = np.random.default_rng(42)
t   = np.linspace(0, 365, 366)

# Synthetic seasonal hydrograph (Ebro-style)
Q = (150
     + 230 * np.exp(-((t - 105) / 45)**2)
     + 40  * np.cos(2 * np.pi * (t - 200) / 365)
     + rng.normal(0, 12, len(t)))

fig, ax = plt.subplots(figsize=(10, 4))

# Uncertainty band
ax.fill_between(t, Q - 40, Q + 40, color="#58a6ff", alpha=0.10, label="±1σ band")
# Main series
ax.plot(t, Q, color="#58a6ff", lw=1.4, label="Q (m³/s)")
# Mean reference
ax.axhline(Q.mean(), color="#8b5cf6", lw=1.0, ls="--", label=f"Mean {Q.mean():.0f} m³/s")

ax.set_xlabel("Day of year")
ax.set_ylabel("Discharge  (m³/s)")
ax.set_title("Annual Hydrograph — Ebro Basin", loc="left", pad=12)
ax.xaxis.set_major_locator(ticker.MultipleLocator(60))
ax.legend(framealpha=0.15, edgecolor="#3c494c")
ax.grid(True)
ax.spines[["top", "right"]].set_visible(False)

plt.tight_layout()
plt.savefig("hydrograph.pdf")   # vector for journal
plt.savefig("hydrograph.png", dpi=300)  # raster for presentation

Interactive Time Series

For web figures, Plotly's graph_objects API gives full control over every trace property. The pattern below builds a dual-axis figure combining annual precipitation anomalies for Spain (1950–2024) with the NAO index — a teleconnection that drives much of the interannual variability in western European rainfall.

02_plotly_timeseries.py 
import plotly.graph_objects as go
import numpy as np, pandas as pd

# ── Data ────────────────────────────────────────────────────────
years = pd.RangeIndex(1950, 2025)
rng   = np.random.default_rng(42)

trend   = -0.22 * (years - 1950) / 10
decadal = 1.4  * np.sin((years - 1950) * 2 * np.pi / 28)
p_anom  = trend + decadal + rng.normal(0, 1.6, len(years))
nao_idx = -0.6 * p_anom + rng.normal(0, 0.9, len(years))
rolling = pd.Series(p_anom).rolling(5, center=True).mean()

# ── Figure ──────────────────────────────────────────────────────
fig = go.Figure()

# Bar chart: annual anomaly (positive=blue, negative=purple)
colors = ["rgba(88,166,255,0.55)" if v >= 0 else "rgba(139,92,246,0.55)"
          for v in p_anom]
fig.add_trace(go.Bar(
    x=years, y=p_anom, name="P anomaly (mm/month)",
    marker_color=colors, yaxis="y1"))

# 5-year rolling mean
fig.add_trace(go.Scatter(
    x=years, y=rolling, name="5-yr mean", mode="lines",
    line=dict(color="#ffffff", width=2.2), yaxis="y1"))

# NAO index on secondary axis
fig.add_trace(go.Scatter(
    x=years, y=nao_idx, name="NAO index", mode="lines",
    line=dict(color="#8b5cf6", width=1.4, dash="dot"), yaxis="y2"))

fig.update_layout(
    template="plotly_dark",
    paper_bgcolor="#030e20", plot_bgcolor="#0a1628",
    font=dict(family="JetBrains Mono", color="#d7e3fc"),
    yaxis=dict(title="P anomaly (mm/month)", gridcolor="#1f2a3d"),
    yaxis2=dict(title="NAO index", overlaying="y", side="right"),
    xaxis=dict(rangeslider=dict(visible=True)),
    hovermode="x unified",
    legend=dict(bgcolor="rgba(3,14,32,0.8)", bordercolor="#3c494c"),
)
Live — Spain P anomaly + NAO 1950–2024

The negative correlation between NAO and Spanish precipitation is visible across most decades: strong positive NAO phases (upper-right in the secondary axis) coincide with dry anomalies (purple bars). Use the range slider to isolate decadal windows.

Return Level Plot

The return level plot is one of the most important diagnostics in extreme value analysis. It should show: (1) the fitted GEV curve, (2) its confidence interval as a shaded band, and (3) the empirical data plotted at their Gringorten plotting positions. If points fall outside the CI band, the model is poorly specified.

03_return_level_plot.py 
import plotly.graph_objects as go
import numpy as np

# GEV parameters from pyextremes fit (see EVT tutorial)
mu, sigma, xi = 178.34, 62.17, -0.0812
n_obs = 50    # 50 years of annual maxima

def gev_rl(T, mu, sigma, xi):
    """Return level for return period T (years)."""
    p = 1 - 1 / T
    z = -np.log(p)
    return mu + sigma / xi * (z**(-xi) - 1)

# GEV curve over a dense T grid
T_grid = np.logspace(np.log10(1.01), np.log10(500), 200)
RL     = gev_rl(T_grid, mu, sigma, xi)
CI_lo  = gev_rl(T_grid, mu - 1.96*8.4, sigma - 1.96*4.1, xi)
CI_hi  = gev_rl(T_grid, mu + 1.96*8.4, sigma + 1.96*4.1, xi)

# Empirical Gringorten plotting positions
rng    = np.random.default_rng(42)
am     = np.sort(gev_rl(
    1/(rng.uniform(0.02, 0.98, n_obs)), mu, sigma, xi)
    + rng.normal(0, 15, n_obs))
i_rank = np.arange(1, n_obs + 1)
pi     = (i_rank - 0.44) / (n_obs + 0.12)
T_emp  = 1 / (1 - pi)

fig = go.Figure()
fig.add_trace(go.Scatter(    # CI band upper edge (hidden)
    x=np.concatenate([T_grid, T_grid[::-1]]),
    y=np.concatenate([CI_hi, CI_lo[::-1]]),
    fill="toself", fillcolor="rgba(88,166,255,0.12)",
    line_color="rgba(0,0,0,0)", name="95% CI", showlegend=True))
fig.add_trace(go.Scatter(    # GEV fitted line
    x=T_grid, y=RL, mode="lines", name="GEV fit",
    line=dict(color="#58a6ff", width=2.2)))
fig.add_trace(go.Scatter(    # empirical points
    x=T_emp, y=am, mode="markers", name="Empirical (Gringorten)",
    marker=dict(color="#d7e3fc", size=5, symbol="circle-open")))

fig.update_layout(
    xaxis=dict(type="log", title="Return period (yr)", gridcolor="#1f2a3d"),
    yaxis=dict(title="Return level (m³/s)", gridcolor="#1f2a3d"),
    paper_bgcolor="#030e20", plot_bgcolor="#0a1628",
    font=dict(family="JetBrains Mono", color="#d7e3fc"))

Teleconnection Heatmap

Annotated heatmaps are the standard way to visualize correlation matrices between climate indices and seasonal precipitation. The key design decision is color scale choice: use a diverging scale (RdBu or equivalent) centered at zero — never a sequential scale for correlation data.

04_teleconnection_heatmap.py 
import plotly.graph_objects as go
import numpy as np

indices = ["NAO", "WeMO", "EA", "AMO", "ENSO", "PDO"]
seasons = ["DJF", "MAM", "JJA", "SON"]

# Pearson r: teleconnection index vs Spanish basin-average precipitation
corr = np.array([
    [-0.65, -0.38, -0.09, -0.31],   # NAO
    [ 0.58,  0.31,  0.07,  0.44],   # WeMO
    [ 0.43,  0.27,  0.06,  0.35],   # EA
    [ 0.27,  0.14, -0.05,  0.19],   # AMO
    [-0.32, -0.19, -0.04, -0.25],   # ENSO
    [-0.21, -0.13,  0.03, -0.16],   # PDO
])

annotations = [
    dict(x=j, y=i, text=f"{corr[i,j]:.2f}",
         showarrow=False, font=dict(color="white" if abs(corr[i,j])>0.4 else "#d7e3fc", size=11))
    for i in range(len(indices)) for j in range(len(seasons))
]

fig = go.Figure(go.Heatmap(
    z=corr, x=seasons, y=indices,
    colorscale=[[0,"#8b1a1a"],[0.5,"#1f2a3d"],[1,"#1a4a7a"]],
    zmin=-1, zmax=1,
    colorbar=dict(title="Pearson r", thickness=14),
))
fig.update_layout(annotations=annotations,
    paper_bgcolor="#030e20", plot_bgcolor="#0a1628",
    font=dict(family="JetBrains Mono", color="#d7e3fc"))

Spatial Mapping

For spatial data, Cartopy is the standard in academic Python. It handles coordinate reference systems, geographic features, and projections natively. Below, the static Cartopy code is paired with a live Plotly Scattergeo showing a synthetic Spanish precipitation station network — the same spatial dataset rendered interactively.

05_cartopy_map.py 
import cartopy.crs as ccrs
import cartopy.feature as cfeature
import matplotlib.pyplot as plt
import numpy as np

# Station data: lat, lon, mean annual precip (mm/yr)
stations = {
    "Santiago":   (42.90, -8.54, 1850),
    "Bilbao":     (43.26, -2.93, 1180),
    "Pamplona":   (42.82, -1.65,  680),
    "Zaragoza":   (41.65,  0.88,  320),
    "Madrid":     (40.42, -3.70,  436),
    "Barcelona":  (41.39,  2.18,  640),
    "Valencia":   (39.47, -0.38,  450),
    "Sevilla":    (37.39, -5.99,  559),
    "Murcia":     (37.99, -1.13,  303),
    "Almería":    (36.84, -2.46,  196),
}
lats  = [v[0] for v in stations.values()]
lons  = [v[1] for v in stations.values()]
prec  = [v[2] for v in stations.values()]

proj = ccrs.PlateCarree()
fig, ax = plt.subplots(1, 1, figsize=(8, 6),
                       subplot_kw={"projection": ccrs.Mercator()})
ax.set_extent([-10, 5, 35, 45], crs=proj)

ax.add_feature(cfeature.LAND,  facecolor="#0a1628")
ax.add_feature(cfeature.OCEAN, facecolor="#030e20")
ax.add_feature(cfeature.COASTLINE, edgecolor="#3c494c", lw=0.8)
ax.add_feature(cfeature.BORDERS, linestyle="--", edgecolor="#3c494c", lw=0.5)

sc = ax.scatter(lons, lats, c=prec, s=80,
               cmap="Blues", vmin=150, vmax=1900,
               edgecolors="white", linewidths=0.5,
               transform=proj, zorder=5)
plt.colorbar(sc, ax=ax, label="MAP (mm/yr)", shrink=0.7)
ax.set_title("Mean Annual Precipitation — Spain", loc="left")
Live — Precipitation Station Network (Spain)

The north–south precipitation gradient is striking: the Cantabrian coast receives 5–9× more rainfall than the semi-arid southeast. Hover over each station for exact values. This gradient is driven by Atlantic storm tracks and the rain-shadow effect of the Pyrenees and Iberian System.
